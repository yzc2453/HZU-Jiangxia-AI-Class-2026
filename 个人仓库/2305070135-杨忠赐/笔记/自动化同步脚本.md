# sync_fork.sh脚本代码

#!/bin/bash
echo "=== 同步Fork仓库 ==="
cd ~/Desktop/HZU-Jiangxia-AI-Class-2026

#1. 切换到main分支
git checkout main

#2. 拉取上游更新
echo "拉取老师的最新修改..."
git fetch upstream

#3. 合并
git merge upstream/main

#4. 推送到你的GitHub
git push origin main

echo "✅ 同步完成！你的Fork已更新"
echo "当前状态："
git log --oneline -3 --graph


# 使用方法

    #给执行权限
    chmod +x sync_fork.sh

    #运行
    ./sync_fork.sh

# 检查是否需要同步
    #查看本地和上游的差异
    git fetch upstream
    git log --oneline main..upstream/main
